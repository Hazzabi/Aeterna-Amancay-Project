Configuración de Switches Multicapa
SW-Dist-Alborada
text
enable
configure terminal
hostname SW-Dist-Alborada
ip routing

vlan 10
 name Vision-Armonia
vlan 20
 name Eco-Taller
vlan 99
 name Guardianes-IT

interface Vlan10
 ip address 172.20.1.1 255.255.255.224
 ip helper-address 172.20.3.10
 no shutdown

interface Vlan20
 ip address 172.20.1.33 255.255.255.224
 ip helper-address 172.20.3.10
 no shutdown

interface Vlan99
 ip address 172.20.1.65 255.255.255.240
 ip helper-address 172.20.3.10
 no shutdown

interface GigabitEthernet0/1
 no switchport
 ip address 172.20.4.2 255.255.255.252
 no shutdown

interface range FastEthernet0/1-2
 switchport mode trunk
 switchport trunk allowed vlan 10,20,99
 no shutdown

interface range FastEthernet0/3-6
 switchport mode access
 switchport access vlan 99
 switchport port-security
 switchport port-security maximum 1
 switchport port-security violation shutdown
 no shutdown

router ospf 1
 router-id 1.1.1.2
 network 172.20.1.0 0.0.0.31 area 0
 network 172.20.1.32 0.0.0.31 area 0
 network 172.20.1.64 0.0.0.15 area 0
 network 172.20.4.0 0.0.0.3 area 0
 passive-interface default
 no passive-interface GigabitEthernet0/1

end
write memory
